#!/bin/bash
# TensorFlow 2.21 GPU Installer for WSL2 + VS Code

set -e

if [ "$EUID" -eq 0 ]; then
  echo "Do NOT run as root"
  exit 1
fi

echo "Updating system..."
sudo apt update
sudo apt install -y build-essential wget curl git python3-dev

# Install Miniconda if missing
if [ ! -d "$HOME/miniconda" ]; then
  wget -O miniconda.sh \
  https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
  bash miniconda.sh -b -p $HOME/miniconda
fi

export PATH="$HOME/miniconda/bin:$PATH"
source "$HOME/miniconda/etc/profile.d/conda.sh"

conda init bash
conda config --set auto_activate_base false
conda update -n base -c defaults conda -y

# Create environment if missing
if conda env list | grep -q ".tf221"; then
  echo "Environment exists"
else
  conda create -n .tf221 python=3.11 -y
fi

conda activate .tf221

pip install --upgrade pip

# TensorFlow GPU
pip install "tensorflow[and-cuda]==2.21.*"

# ML libraries
pip install numpy pandas matplotlib seaborn

# VS Code kernel support
pip install ipykernel

# Optional GPU tools
pip install pycuda tensorrt

echo ""
echo "Testing GPU..."

python - <<EOF
import tensorflow as tf
print("TensorFlow:", tf.__version__)
print("GPUs:", tf.config.list_physical_devices('GPU'))
EOF

echo ""
echo "Done!"
echo "Activate later with:"
echo "conda activate .tf221"
